  <script>
    // Header behavior
    const header = document.getElementById('header');
    const menuToggle = document.getElementById('menuToggle');
    const mobileOverlay = document.getElementById('mobileOverlay');
    const logo = document.querySelector('.logo');
    const body = document.body;

    window.addEventListener('scroll', () => {
      header.classList.toggle('scrolled', window.scrollY > 50);
    });

    function closeMenu() {
      mobileOverlay.classList.remove('active');
      menuToggle.classList.remove('active');
      header.classList.remove('menu-open');
      logo.classList.remove('hide');
      body.style.overflow = '';
    }
    function openMenu() {
      mobileOverlay.classList.add('active');
      menuToggle.classList.add('active');
      header.classList.add('menu-open');
      logo.classList.add('hide');
      body.style.overflow = 'hidden';
    }
    if (menuToggle) {
      menuToggle.addEventListener('click', () => {
        mobileOverlay.classList.contains('active') ? closeMenu() : openMenu();
      });
    }
    document.querySelectorAll('.mobile-nav a').forEach(link => link.addEventListener('click', closeMenu));
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Escape' && mobileOverlay.classList.contains('active')) closeMenu();
    });
    mobileOverlay.addEventListener('touchmove', (e) => {
      if (e.target === mobileOverlay) e.preventDefault();
    }, { passive: false });

    // GSAP Horizontal Scroll
    gsap.registerPlugin(ScrollTrigger);

    const pin = document.getElementById('projectsPin');
    const track = document.getElementById('projectsTrack');
    const progressBar = document.getElementById('scrollProgress');
    const filterBar = document.querySelector('.filter-bar');
    const prefersReduced = window.matchMedia('(prefers-reduced-motion: reduce)');

    function headerOffset() {
      return document.getElementById('header')?.offsetHeight || 80;
    }

    let horizTween;

    function setupHorizontalScroll() {
      // kill existing timeline/trigger
      if (horizTween) horizTween.kill();
      ScrollTrigger.getById('projectsST')?.kill();

      // clear transforms
      gsap.set(track, { clearProps: 'transform' });

      if (prefersReduced.matches) {
        if (progressBar) progressBar.style.width = '0%';
        return;
      }

      horizTween = gsap.to(track, {
        x: () => -(track.scrollWidth - pin.offsetWidth),
        ease: 'none',
        scrollTrigger: {
          id: 'projectsST',
          trigger: pin,
          start: () => 'top+=' + headerOffset() + ' top',
          end: () => '+=' + Math.max(0, track.scrollWidth - pin.offsetWidth),
          pin: true,
          scrub: true,
          anticipatePin: 1,
          invalidateOnRefresh: true,
          onUpdate: (self) => {
            if (progressBar) progressBar.style.width = (self.progress * 100).toFixed(2) + '%';
          }
        }
      });
    }

    // Initial setup
    window.addEventListener('load', () => {
      setupHorizontalScroll();
      ScrollTrigger.refresh();
    });
    window.addEventListener('resize', () => ScrollTrigger.refresh());
    if (prefersReduced.addEventListener) {
      prefersReduced.addEventListener('change', () => {
        setupHorizontalScroll();
        ScrollTrigger.refresh();
      });
    } else if (prefersReduced.addListener) {
      // Safari fallback
      prefersReduced.addListener(() => {
        setupHorizontalScroll();
        ScrollTrigger.refresh();
      });
    }

    // Refresh when images load (for accurate widths)
    document.querySelectorAll('#projectsTrack img').forEach(img => {
      img.addEventListener('load', () => ScrollTrigger.refresh(), { once: true });
    });

    // Filter behavior — refresh ScrollTrigger so distances recalc
    if (filterBar) {
      filterBar.addEventListener('click', (e) => {
        const btn = e.target.closest('.filter-btn');
        if (!btn) return;

        const filter = btn.dataset.filter;

        // Toggle button states
        filterBar.querySelectorAll('.filter-btn').forEach(b => {
          const isActive = b === btn;
          b.classList.toggle('active', isActive);
          b.setAttribute('aria-pressed', isActive ? 'true' : 'false');
          b.setAttribute('aria-selected', isActive ? 'true' : 'false');
        });

        // Show/hide cards
        document.querySelectorAll('#projectsTrack .project-card').forEach(card => {
          const cats = (card.dataset.category || '').split(' ');
          const show = filter === 'all' || cats.includes(filter);
          card.hidden = !show;
        });

        // Rebuild scroll distances
        setupHorizontalScroll();
        requestAnimationFrame(() => ScrollTrigger.refresh());
      });
    }
  </script>
